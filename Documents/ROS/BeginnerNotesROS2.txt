# Setting up ROS enviroment in Ubuntu 20.04

# Setup enviroment:
$ sudo apt install build-essential gcc make perl dkms
$ sudo apt install terminator 	#Terminator
$ sudo snap install --classic code 	#VSCode
	# VScode Toolboxes: c/c++(Microsoft) CMake(twxs) Python(Microsoft)
$ sudo apt install python3-pip 	#python pip


# install ROS: https://index.ros.org/doc/ros2/Installation/Foxy/Linux-Install-Debians/
# locale
$ sudo apt update && sudo apt install locales
$ sudo locale-gen en_US en_US.UTF-8
$ sudo update-locale LC_ALL=en_US.UTF-8 LANG=en_US.UTF-8
$ export LANG=en_US.UTF-8

# Sources 
$ sudo apt update && sudo apt install curl gnupg2 lsb-release
$ curl -s https://raw.githubusercontent.com/ros/rosdistro/master/ros.asc | sudo apt-key add -
$ sudo sh -c 'echo "deb [arch=$(dpkg --print-architecture)] http://packages.ros.org/ros2/ubuntu $(lsb_release -cs) main" > /etc/apt/sources.list.d/ros2-latest.list'

$ sudo apt update
$ sudo apt install ros-foxy-desktop

# argcomplete
$ pip3 install -U argcomplete

# Source enviroment
$ source /opt/ros/foxy/setup.bash 	#add to .bashrc

# Creating packages with Colcon: https://index.ros.org/doc/ros2/Tutorials/Colcon-Tutorial/
$ sudo apt install python3-colcon-common-extensions

# enable autocomplete Colcon
$ source /usr/share/colcon_argcomplete/hook/colcon-argcomplete.bash 	#add to .bashrc

# Create Workspace https://index.ros.org/doc/ros2/Tutorials/Colcon-Tutorial/
$ mkdir -p ~/ros2_ws/src
$ cd ~/ros2_ws
$ colcon build
$ source ~/ros2_ws/install/setup.bash 	#source workspace, add to .bashrc

# create Package https://index.ros.org/doc/ros2/Tutorials/Creating-Your-First-ROS2-Package/
# Python
$ ros2 pkg create my_py_pkg --build-type ament_python --dependencies rclpy
$ colcon build --packages-select my_py_pkg

# CPP
$ ros2 pkg create my_cpp_pkg --build-type ament_cmake --dependencies rclcpp
$ colcon build --packages-select my_py_pkg
